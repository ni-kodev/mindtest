SELECT 
Article.Name AS ArticleName
, Tag.Name AS TagName
FROM 
ArticleTag 
INNER JOIN Tag ON ArticleTag.TagId = Tag.Id
RIGHT OUTER JOIN Article ON ArticleTag.ArticleId = Article.Id
